function addLanguageInURL (url) { 
		if(url.toLowerCase().indexOf("languageid=") == -1){
			url = url + "&LanguageId="; 
		}    
		var params = url.split("&");               
		for (var i = 0; i < params.length; i++) {   
			var pair = params[i].split("=");          
				if (pair[0] == "LanguageId") {
						params[i] = pair[0] + "=" + LanguageId;
					}     
				}   
		url = params.join("&");   
		self.location.href = url;
	}
